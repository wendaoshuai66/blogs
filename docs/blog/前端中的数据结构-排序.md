function quickSort(arr) {
    if (arr.length === 0) {
        return []
    }
    let base = arr[0],
        less = [],
        big = [];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > base) {
            big.push(arr[i])
        } else {
            less.push(arr[i])
        }
    }
    return quickSort(less).concat(base, quickSort(big))
}
